find_package(Magnum REQUIRED GL GlfwApplication)

set_directory_properties(PROPERTIES CORRADE_USE_PEDANTIC_FLAGS ON)

add_executable(YAGE main.cpp)
target_link_libraries(YAGE PRIVATE
    Magnum::Application
    Magnum::GL
    Magnum::Magnum)

set_property(DIRECTORY ${PROJECT_SOURCE_DIR} PROPERTY VS_STARTUP_PROJECT YAGE)
